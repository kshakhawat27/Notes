--add possible columns to table PRA_MN_RPT_TAB_XL_PD
DECLARE @INIT INT =61, @COLUMN NVARCHAR(10) ='';

WHILE(@INIT<=100)
BEGIN
	SET @COLUMN=CAST(@INIT AS VARCHAR);
	BEGIN TRY
		IF NOT EXISTS(SELECT TOP 1 1 FROM INFORMATION_SCHEMA.COLUMNS WHERE  [TABLE_NAME] = 'PRA_MN_RPT_TAB_XL_PD' AND [COLUMN_NAME] = 'loan_'+@COLUMN)
		BEGIN
			EXEC(
			'ALTER TABLE pksf.PRA_MN_RPT_TAB_XL_PD
			 ADD loan_'+@COLUMN+' DECIMAL(17,2) NULL DEFAULT(0)'
			);
		END
	END TRY
	BEGIN CATCH
	END CATCH
	SET @INIT=@INIT+1;
END
